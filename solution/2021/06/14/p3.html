n = 600851475143

def is_prime(n):
    for i in range(3, int(n ** 0.5) + 1, 2):
        if n % i is 0:
            return False
    return True

max = 0

for i in range(3, n, 2):
    if i % 3 is 0 or i % 5 is 0 or i % 7 is 0:
        continue
    
    if n % i is 0:
        if is_prime(i):
            max = i

print(max)

How to do prime factorization with division 

Prime factors | Remainder

            2 |        10 (= 20 / 2)
            5 |         5 (= 10 / 2)
              |         1 (= 5 / 5)

Pseudocode 

while n > 1:
    find number i that makes n % i == 0:
        n = n / i
        restart i from 2
n = 600851475143

max = 0
i = 2

while n > 1:
    if n % i is 0:  # prime factor found
        n = n // i
        if i > max:
            max = i
        i = 2
        continue
    
    i += 1

print(max)
