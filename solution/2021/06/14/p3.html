Problem

The prime factors of 13195 are 5, 7, 13 and 29.

What is the largest prime factor of the number 600851475143 ?

Initial idea

n = 600851475143

def is_prime(n):
    for i in range(3, int(n ** 0.5) + 1, 2):
        if n % i is 0:
            return False
    return True

max = 0

for i in range(3, n, 2):
    if i % 3 is 0 or i % 5 is 0 or i % 7 is 0:
        continue
    
    if n % i is 0:
        if is_prime(i):
            max = i

print(max)

while n > 1:
